public virtual class Vehicle {
    
    public Integer amountVehicles = 8;
    public String vehicleSpeed = 'Full Speed';
    public String direction = 'north';
    public String stopping;
    public enum Transport {
        AIR,
        WATER,
        LAND
    }
    
    // method to determine movement
    public void Movement(String speed, String direction) {
       this.vehicleSpeed = speed;
       this.direction = direction; 
      }
    
    // method to determine how fast it stops
    public void Stopped(String howFast) {
        this.stopping = howFast;
    }
   	
    // how big is the fleet
    public void FleetChange(integer amount) {
        this.amountVehicles = amount;
    }
    
    
    
}
