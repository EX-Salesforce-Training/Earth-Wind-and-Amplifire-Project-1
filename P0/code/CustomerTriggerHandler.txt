public class CustomerTriggerHandler {
    
    // check if customer added an email
    // if not add a default one
    public static void CheckforEmail(List<Customer__c> customertoCheck){
        List<Customer__c> toUpdate = new List<Customer__c>();
        
        for(Customer__c isEmail : customertoCheck){
            isEmail = [SELECT Customer_Email__c FROM Customer__c];
            if(isEmail.Customer_Email__c == null){
             	isEmail.Customer_Email__c = 'default@.com';
                toUpdate.add(isEmail);
            }
        }
        database.update(toUpdate);
    }
    
    // check if customer added an order
    // if not alert them to it
    public static void CheckforOrder(List<Customer__c> orderToCheck){
        
        for(Customer__c check : orderToCheck) {
            check = [SELECT Food_Order__c FROM Customer__c];
            if(check.Food_Order__c == null){
                check.addError('ERROR: You did not place an order.');
            }
        }
        
    }

}
